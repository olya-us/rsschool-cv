Code example

This code is handling the acceptance of cookies on a website.


let cookieAcceptBtn = document.getElementById('cookieAccept');

if(cookieAcceptBtn) {
    cookieAcceptBtn.addEventListener('click', function() {
        saveCookies();
    })
}

function saveCookies() {
    let currentDate = new Date();
    let expirationDate = currentDate;
    expirationDate.setMonth(currentDate.getMonth() + 6);
    document.cookie = "cookieConsentAccepted=1; expires=" + expirationDate.toUTCString() + "; path=/";

    hideModalCookies();
}

function hideModalCookies() {
    let modalCookie = document.querySelector('.modal-cookies');
    modalCookie.style.display = 'none';
}
